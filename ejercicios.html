def bfs(grafo, inicio):

def contar_componentes(grafo):
  visitado = set()
  componentes = 0

  def dfs(nodo):
    visitado.add(nodo)
    for vecino in grafo[nodo]:
      if vecino not in visitado:
        dfs(vecino)

  for nodo in grafo:
    if nodo not in visitado:
      dfs(nodo)
      componentes += 1

  return componentes
